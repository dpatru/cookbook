\section{Thai Peanut Quinoa Salad}
\begin{recipe}
	\yield{? servings}
	\cooktime{15-30 minutes}
	
	\pre{This interesting quinoa salad is great warm and fresh or straight from the refrigerator.}
	
	\ingredients{
		\sfrac{3}{4}	& uncooked quinoa \\
		1\sfrac{1}{2}	& \cup water \\
		2				& \cup shredded purple cabbage \\
		1				& \cup grated carrot \\
		1				& \cup sliced snap peas \\
		\sfrac{1}{2}	& \cup chopped cilantro \\
		\sfrac{1}{4}	& \cup sliced green onions \\
		\sfrac{1}{4}	& \cup chopped, roasted, salted peanuts \\
	}
	
	Cook quinoa. Shred cabbage, grate carrots, slice snap peas, chop cilantro, slice green onions, and chop peanuts. Set aside.
	
	\textbf{Sauce:}	
		
	\ingredients{
		\sfrac{1}{4}	& \cup creamy peanut butter \\
		3				& \T soy sauce / tamari \\
		1				& \T syrup or honey \\
		1				& \T rice vinegar \\
		1				& \t sesame oil \\
		1				& \t ginger \\
		\sfrac{1}{2}	& lime, juiced \\
		pinch			& red pepper flakes \\
	}

	Whisk together peanut butter and soy sauce over medium heat. Once smooth, add remaining ingredients and whisk until combined.

	In a large bowl, combine quinoa, sauce, and vegetables. Toss to combine. Garnish with peanuts and cilantro to serve.
	
\end{recipe}
