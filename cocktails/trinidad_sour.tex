\section{Trinidad Sour}
\begin{recipe}
	\source{Giuseppe Gonzalez}
	
	\pre{A paradigm shifter with its base of Angostura bitters.}
	
	\ingredients{
		1				& \oz Angostura bitters \\
		1				& \oz orgeat \\
		\sfrac{3}{4}	& \oz fresh lemon juice \\
		\sfrac{1}{2}	& \oz rye whiskey (100-proof)
	}
	
	Combine ingredients in a shaker and fill with ice. Shake well to chill; strain into a chilled glass.
	
\end{recipe}
